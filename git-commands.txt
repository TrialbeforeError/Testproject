# local clone of github project
git clone <remote address>

# add file to your repository
git add <file> 

# gives you the status
git status

# commit changes in all files
git commit -a -m

# push your committed changes to the original
git push
git push origin master


#show log of changes 
git log -10

# configure your github account
git config --global user.name "<name>"

# configure your github emailaccount
git config --global user.email "<email-address>"

# list your git configuration
git config --list
